import java.awt.Desktop; import java.io.File; import java.io.FileWriter; import java.io.IOException; public class Mesaj1Martie { private static final String HTML_TEMPLATE = """ La mulți ani de 1 Martie 🌸

🌸 La mulți ani de 1 Martie! 🌸

"""; public static void main(String[] args) { try { File file = createHtmlFile("Mesaj_1_Martie_Animat.html", HTML_TEMPLATE); openInBrowser(file); } catch (IOException e) { System.err.println("Eroare: " + e.getMessage()); } } private static File createHtmlFile(String fileName, String content) throws IOException { File file = new File(System.getProperty("java.io.tmpdir"), fileName); try (FileWriter writer = new FileWriter(file, false)) { writer.write(content); } return file; } private static void openInBrowser(File file) throws IOException { if (Desktop.isDesktopSupported()) { Desktop.getDesktop().browse(file.toURI()); System.out.println("Mesajul animat de 1 Martie s-a deschis în browser. 🌸"); } } }
